```python
# In _attach_teacher_signals, ROD_PT branch
if mode == "ROD_PT":
    position_batch = _build_position_teacher_batch(...)  # §3.6 mirror R5
    if position_batch is None:
        # No META emit — w_position = 1.0 for all (no amplify)
        data.batch["sdc_position_log_prob_meta"] = torch.zeros(B, device=target_device)
    else:
        position_batch.meta_info["temperature"] = rollout_temp
        pos_out = trainer._compute_ref_log_prob(position_batch)

        # Map back to full batch (rollouts without META → log_prob = 0 = no amplify)
        full_log_prob_meta = torch.zeros(B, device=target_device)

        data.batch["sdc_position_log_prob_meta"] = full_log_prob_meta
```

`compute_sdc_gdpo_advantage` (verl_sdc_utils.py)에 ROD_PT mode dispatch 추가:
```python
# Existing: w_attr = clip(exp(sign × attr_log), 1-ε, 1+ε)  (R5 메타 내용)
# NEW for ROD_PT:
log_prob_meta = batch["sdc_position_log_prob_meta"].to(device)  # [B]
clip_eps = float(config.get("sdc_clip_eps_w", 0.2))
sign_per_seq = torch.sign(seq_adv).squeeze(-1)  # [B]
w_position = torch.clamp(
    torch.exp(sign_per_seq * log_prob_meta), 1.0 - clip_eps, 1.0 + clip_eps
)  # [B] in [0.8, 1.2]
# Apply on advantage at meta region (positional signal applies where META is emitted)
advantages = base_advantages * w_attr * meta_mask + base_advantages * w_position.unsqueeze(1) * meta_mask + base_advantages * (1.0 - meta_mask)
# 또는 w_combined = w_attr * w_position on meta region (둘 다 RLSD pattern)
```

**RLSD invariant 보존**:
- `sign × log_prob_meta`: prob_meta=0.95 → log=-0.05 → sign=+1이면 exp(-0.05)=0.95 (작은 amplify)
- prob_meta=0.05 → log=-3.0 → exp(sign × -3) = 0.05 (clip → 0.8로 saturate)
- 부호는 sign 변수가 보존, magnitude만 modulate.

### 3.6 `_build_position_teacher_batch` (round 4 R1 fix — R5 mirror)

**v5.14 결함**: `prompts=input_ids[:, :-1]`/`responses=input_ids[:, -1:]` splice는 right-pad 후 PAD 위치 위험.

**v5.15 fix — verl_sdc.py:386-409 정확 mirror**:

```python
def _build_position_teacher_batch(
    *, tokenizer, prompt_texts, gold_answers, responses, response_mask,
    meta_start_id, rollout_temp, max_response_length=4,  # META + few buffer
):
    """R5 _build_teacher_logprob_batch (verl_sdc.py:335-420) mirror.

    Key invariant: prompt = left-padded full text up to META insertion point,
                  response = META_START token only (length 1, right-side).
    """
    B = responses.size(0)

    # Step 1: text prep — find first META_START position p, build prompt text
    prompt_only_texts: list[str] = []  # without META appended
    rollout_indices: list[int] = []
    for b in range(B):
        valid = (responses[b] == meta_start_id) & response_mask[b].bool()
        nz = valid.nonzero(as_tuple=True)[0]
        if nz.numel() == 0:
            continue
        p = int(nz[0].item())
        if p == 0:
            comp_text = ""
        else:
            comp_ids = responses[b, :p][response_mask[b, :p].bool()]
            comp_text = tokenizer.decode(comp_ids, skip_special_tokens=False)
        text = f"{prompt_texts[b]} Answer: {gold_answers[b]} {comp_text}"
        prompt_only_texts.append(text)
        rollout_indices.append(b)

    if not prompt_only_texts:
        return None

    # Step 2: tokenize prompts with LEFT padding (R5 verl_sdc.py:391 same convention)
    prev_padding = tokenizer.padding_side
    prev_truncation = tokenizer.truncation_side
    tokenizer.padding_side = "left"
    tokenizer.truncation_side = "left"  # preserve recent context (META 직전)
    try:
        prompt_enc = tokenizer(
            prompt_only_texts,
            add_special_tokens=False,
            padding="max_length",
            truncation=True,
            max_length=4095,  # leave room for META
            return_tensors="pt",
        )
    finally:
        tokenizer.padding_side = prev_padding
        tokenizer.truncation_side = prev_truncation

    prompt_ids = prompt_enc["input_ids"]                    # [N, 4095] left-padded
    prompt_attn = prompt_enc["attention_mask"]              # [N, 4095]
    N = prompt_ids.size(0)

    # Step 3: append META as response (length 1)
    response_ids = torch.full((N, 1), meta_start_id, dtype=prompt_ids.dtype)
    response_attn = torch.ones((N, 1), dtype=prompt_attn.dtype)
    response_mask_out = response_attn.float()

    # Step 4: combined view + position_ids (R5 verl_sdc.py:399-409 pattern)
    input_ids = torch.cat([prompt_ids, response_ids], dim=1)        # [N, 4096]
    attention_mask = torch.cat([prompt_attn, response_attn], dim=1)  # [N, 4096]
    # left-padded prompt → position_ids continuous from real start
    position_ids = (attention_mask.cumsum(dim=1) - 1).clamp(min=0)

    return DataProto.from_dict(
        tensors={
            "prompts": prompt_ids,                # [N, 4095] left-padded
            "responses": response_ids,            # [N, 1] = META_START
            "input_ids": input_ids,               # [N, 4096]
            "attention_mask": attention_mask,     # [N, 4096]
            "response_mask": response_mask_out,   # [N, 1]
            "position_ids": position_ids,         # [N, 4096]
        },
        non_tensors={"rollout_indices": np.array(rollout_indices, dtype=np.int64)},
        meta_info={"temperature": rollout_temp},
    )
```

**핵심 변경**: `prompt_ids` (left-pad)와 `response_ids` (META 1 token)을 **별도 tokenize 후 concat**. R5 패턴 정확 mirror. PAD가 마지막에 올 위험 0.

```python
def position_factor_from_log_prob(log_prob_meta, sign_advantage, clip_eps=0.2):
    """RLSD-style multiplicative factor for position teacher signal.
    Returns clip(exp(sign × log_prob_meta), 1-eps, 1+eps).
    """
    return torch.clamp(torch.exp(sign_advantage * log_prob_meta), 1.0 - clip_eps, 1.0 + clip_eps)
```

`tests/test_verl_rod_pt_units.py`:
```python
def test_position_factor_amplify_correct():
    """Positive advantage + high prob_meta → factor > 1.0 (amplify good rollout)."""
    log_prob_meta = torch.tensor([-0.05])  # prob = 0.95
    sign_a = torch.tensor([1.0])  # positive advantage
    f = position_factor_from_log_prob(log_prob_meta, sign_a, clip_eps=0.2)
    assert 1.0 < f.item() <= 1.2  # clip upper

def test_position_factor_dampen_wrong():
    """Negative advantage + high prob_meta → factor < 1.0 (dampen bad rollout)."""
    log_prob_meta = torch.tensor([-0.05])
    sign_a = torch.tensor([-1.0])
    f = position_factor_from_log_prob(log_prob_meta, sign_a, clip_eps=0.2)
    assert 0.8 <= f.item() < 1.0

def test_position_factor_clip_lower():
    """Very low prob_meta → exp blows past 0.8, clipped."""
    log_prob_meta = torch.tensor([-3.0])
    sign_a = torch.tensor([1.0])
    f = position_factor_from_log_prob(log_prob_meta, sign_a, clip_eps=0.2)
    assert f.item() == 0.8  # clip lower

def test_no_meta_emit_no_amplify():
    """When no META emit, log_prob = 0 → factor = exp(0) = 1.0."""
    log_prob_meta = torch.tensor([0.0])
    sign_a = torch.tensor([1.0])
    f = position_factor_from_log_prob(log_prob_meta, sign_a)
    assert f.item() == 1.0  # passthrough

def test_driver_none_handling():
    """Driver assigns zeros to log_prob_meta when no rollout has META — factor = 1.0."""
    full_log_prob = torch.zeros(4)  # all zeros (no META in any rollout)
    sign_a = torch.tensor([1.0, -1.0, 1.0, -1.0])
    f = position_factor_from_log_prob(full_log_prob, sign_a)
    assert torch.all(f == 1.0)  # all passthrough — no penalty when no META emitted
```

## 9. v5.15 self-critic + round 5 trigger

이번 patch에서 발견된 잠재 결함:

1. **§3.4 advantage * w_attr * meta_mask + advantage * w_position * meta_mask**: 두 factor 더하기 → 메타 영역에서 advantage가 (w_attr + w_position) × base. 의도가 "둘 다 amplify"면 multiplicative product (`w_attr * w_position`)이 더 정확. round 5 결정 필요.

2. **§3.4 sign_per_seq broadcast**: `sign_per_seq = sign(seq_adv).squeeze(-1)` shape [B], `w_position.unsqueeze(1)` shape [B,1]. meta_mask [B,T]와 broadcast OK. 검증 필요.

3. **§3.6 max_length=4095 + META 1 token = 4096**: actor model max_seq_len 호환? Qwen3는 4096이 충분하나 longer prompts에서 left-truncation으로 META 직전 잘릴 수 있음.
